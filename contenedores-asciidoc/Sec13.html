Instalación del cliente NFS en los servidores

En todos los servidores master y worker
dnf install -y nfs-utils

Configuración del servidor NFS

Configuración de Selinux en el servidor NFS (lb.s1.example.com)
semanage fcontext -a -t public_content_t "/shared(/.*)?"
restorecon -Rv /shared
Configuración de firewall
firewall-cmd --add-service=nfs --permanent
firewall-cmd --reload

Instalación del NFS Provisioner

export KUBECONFIG=/etc/rancher/k3s/k3s.yaml
kubectl create namespace nfs-provisioner
kubectl config set-context --current --namespace nfs-provisioner
helm repo add nfs-subdir-external-provisioner https://kubernetes-sigs.github.io/nfs-subdir-external-provisioner/

helm install nfs-subdir-external-provisioner nfs-subdir-external-provisioner/nfs-subdir-external-provisioner \
    --set nfs.server=chi1265.itaipu.int \
    --set nfs.path=/mnt/nfs

Definición del StorageClass por defecto

kubectl annotate storageclass local-path storageclass.kubernetes.io/is-default-class-
kubectl annotate storageclass nfs-client storageclass.kubernetes.io/is-default-class=true

Despliegue de una base de datos sin persistencia

Veamos lo que sucede por defecto con una BD sin persistencia

Creación del namespace
kubectl create namespace pruebas6
kubectl config set-context --current --namespace pruebas6
Creación del recurso deployment
mkdir mysql && cd mysql
cat <<EOF > deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  creationTimestamp: null
  labels:
    app: mysql-deployment
  name: mysql-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mysql-deployment
  strategy:
    type: Recreate
  template:
    metadata:
      creationTimestamp: null
      labels:
        app: mysql-deployment
    spec:
      containers:
      - image: quay.io/sclorg/mysql-80-c9s
        name: mysql
        env:
          - name: MYSQL_USER
            value: myuser
          - name: MYSQL_PASSWORD
            value: mypassword
          - name: MYSQL_DATABASE
            value: mydb
        ports:
        - containerPort: 3306
EOF
Despliegue de la aplicación
kubectl apply -f deployment.yaml

# esperar a que los pods terminen de generarse
kubectl get pods -w

Una vez desplegado el contenedor, ingresar la combinación de teclado kbd:[Ctrl+C] para salir del comando

Ingresar al contenedor
# abrir una terminal en el contenedor
kubectl exec -ti deployment/mysql-deployment -- /bin/bash
Abrir sesión de mysql
# dentro del contenedor, abrir una sesion de mysql
mysql -u$MYSQL_USER -p$MYSQL_PASSWORD $MYSQL_DATABASE
Desplegar algunos datos en la BD
use mydb;

create table Person(id int primary key auto_increment, name varchar(50) not null);

insert into Person(name) values ('Pablo Peralta');
insert into Person(name) values ('Hector Alfonso');

select * from Person;

-- salimos de la sesion de mysql
exit

-- salimos del contenedor
exit
Reiniciar el contenedor
# reiniciar el contenedor
kubectl rollout restart deployment/mysql-deployment
Volvemos a ingresar para verificar los datos
# abrir una terminal en el contenedor
kubectl exec -ti deployment/mysql-deployment -- /bin/bash
Abrir sesión de mysql
# dentro del contenedor, abrir una sesion de mysql
mysql -u$MYSQL_USER -p$MYSQL_PASSWORD $MYSQL_DATABASE
Desplegar algunos datos en la BD
use mydb;

show tables;

-- salimos de la sesion de mysql
exit

-- salimos del contenedor
exit

Con esto verificamos que sin un mecanismo de persistencia, los datos se pierden al reiniciar el Deployment.

Agregar persistencia al deployment

Creación del recurso PersistenVolumeClaim
cat <<EOF > pvc.yaml
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: mysql-data
spec:
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 10Gi
EOF

kubectl apply -f pvc.yaml
Modificación de la aplicación para que soporte persistencia
cat <<EOF > deployment-persistent.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  creationTimestamp: null
  labels:
    app: mysql-deployment
  name: mysql-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mysql-deployment
  strategy:
    type: Recreate
  template:
    metadata:
      creationTimestamp: null
      labels:
        app: mysql-deployment
    spec:
      volumes:
        - name: mysql-volume
          persistentVolumeClaim:
            claimName: mysql-data
      containers:
      - image: quay.io/sclorg/mysql-80-c9s
        name: mysql
        env:
          - name: MYSQL_USER
            value: myuser
          - name: MYSQL_PASSWORD
            value: mypassword
          - name: MYSQL_DATABASE
            value: mydb
        ports:
        - containerPort: 3306
        volumeMounts:
          - name: mysql-volume
            mountPath: /var/lib/mysql/data
EOF

kubectl apply -f deployment-persistent.yaml
Verificar el volume persistente
kubectl exec deployment/mysql-deployment -- df -h /var/lib/mysql/data
Ingresar al contenedor
# abrir una terminal en el contenedor
kubectl exec -ti deployment/mysql-deployment -- /bin/bash
Abrir sesión de mysql
# dentro del contenedor, abrir una sesion de mysql
mysql -u$MYSQL_USER -p$MYSQL_PASSWORD $MYSQL_DATABASE
Desplegar algunos datos en la BD

create table Person(id int primary key auto_increment, name varchar(50) not null);

insert into Person(name) values ('Pablo Peralta');
insert into Person(name) values ('Hector Alfonso');

select * from Person;

-- salimos de la sesion de mysql
exit

-- salimos del contenedor
exit
Reiniciar el contenedor
# reiniciar el contenedor
kubectl rollout restart deployment/mysql-deployment
Volvemos a ingresar para verificar los datos
# abrir una terminal en el contenedor
kubectl exec -ti deployment/mysql-deployment -- /bin/bash
Abrir sesión de mysql
# dentro del contenedor, abrir una sesion de mysql

Despliegue de aplicaciones con StatefulSets

Despliegue de base de datos con Helm
helm install mysql bitnami/mysql
Revisión de la persistencia
kubectl get statefulset
kubectl exec -ti statefulset/mysql -- df -h /bitnami/mysql

Despliegue de bases de datos replicadas

Eliminar el entorno actual
helm uninstall mysql
Reinstalar el release con la opción de replicación
helm upgrade --install mysql bitnami/mysql --set architecture="replication" --set secondary.replicaCount=2

Aprovisionamiento estático

Eliminar el deployment y PVC anteriores
kubectl delete -f deployment-persistent.yaml
kubectl delete -f pvc.yaml
En lb, generar el directorio para el PersistentVolume
mkdir -p /shared/static
chmod 777 /shared/static
Generar el recurso PersistentVolume en formato YAML. Desde m1
cat <<EOF > pv.yaml
apiVersion: v1
kind: PersistentVolume
metadata:
  name: mysql-static
spec:
  nfs:
    server: nfs.s1.example.com
    path: /shared/static
  accessModes:
    - ReadWriteMany
    - ReadWriteOnce
    - ReadOnlyMany
  capacity:
    storage: 10Gi
EOF
Generar el PersistentVolumeClaim estático
cat <<EOF > pvc-static.yaml
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: mysql-data
spec:
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 10Gi
  storageClassName: "" (1)
EOF

  1. Este atributo indica que estamos tratando de utilizar un PV Estático

Aplicar el PVC y el Deployment como anteriormente
kubectl apply -f pvc-static.yaml
kubectl apply -f deployment-persistent.yaml
